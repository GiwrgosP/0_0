import tkinter as tk
from tkinter import tix

def test_tix():
    root = tix.Tk()  # Initialize the Tix Tkinter extension
    root.title("Tix Test")

    # Create a frame
    frame = tk.Frame(root, width=300, height=200)
    frame.pack_propagate(False)
    frame.pack()

    # Create a button
    button = tk.Button(frame, text="Hover over me!")
    button.pack(pady=20)

    # Create a Tix balloon (tooltip)
    balloon = tix.Balloon(root)
    balloon.bind_widget(button, balloonmsg="This is a Tix balloon tooltip!")

    root.mainloop()

if __name__ == "__main__":
    test_tix()
